16  /**
17   * Contains some IO utilities that are used by the API.
18   */
19  public class IOUtils {

21      private static final Logger LOGGER = LoggerFactory.getLogger(IOUtils.class);

23      // Thread safe singleton instance of our class.
24      private static final IOUtils INSTANCE = new IOUtils();
25  
26      /**
27       * Returns the thread safe singleton instance of this class.
28       * @return The thread safe singleton instance of this class
29       */
30      public static IOUtils getInstance() {
31          return INSTANCE;
32      }

34      /**
35       * Copy bytes from an <tt>InputStream</tt> to an <tt>OutputStream</tt>.
36       * <p>
37       * This method buffers the input internally, so there is no need to use a <tt>BufferedInputStream</tt>.
38       * @param input  The <tt>InputStream</tt> to read from
39       * @param output The <tt>OutputStream</tt> to write to
40       * @return The number of bytes copied
41       * @throws NullPointerException If the input or output is null
42       * @throws IOException          If an I/O error occurs
43       */
44      public long copy(final InputStream input, final OutputStream output) throws IOException {
45          final byte[] buffer = new byte[8192];
46          long count = 0;
47  
48          int number = input.read(buffer);
49          while (number != -1) {
50              output.write(buffer, 0, number);
51              count += number;
52              number = input.read(buffer);
53          }
54  
55          return count;
56      }
57  
58      /**
59       * Closes the stream and ignores any <tt>IOExceptions</tt> that are thrown.
60       * @param closeable The <tt>Closable</tt> to close
61       */
62      public void closeQuietly(final Closeable closeable) {
63          try {
64              if (closeable != null) {
65                  closeable.close();
66              }
67          } catch (final IOException ioe) {
68              // Ignore...
69              LOGGER.debug("IOException caught while trying to close the stream.", ioe);
70          }
71      }
