49   * communication mechanism. This implementation uses the HttpClient provided by the <a href="http://hc.apache.org/">
50   * Apache HttpComponents</a> project. It defaults to using the HTTP <tt>GET</tt> method, if you wish to send ACI actions

52   * <p>
53   * This implementation of the {@link com.autonomy.aci.client.transport.AciHttpClient} interface does no configuration of
54   * the {@code HttpClient} that it uses. It expects all the configuration to have been done by the user before passing it
55   * to this object. This configuration can be done in normal code, via the

96      /**
97       * Turns the {@code parameters} and {@code serverDetails} into either an HTTP GET or POST request.
98       * @param serverDetails The details of the ACI server the request will be sent to
99       * @param parameters    The parameters to send with the ACI action.
100      * @return A HTTP GET or POST request that can be used to execute the ACI action
101      * @throws EncryptionCodecException     If something went wrong encrypting the parameters
102      * @throws URISyntaxException           If something went wrong creating the URI to send the action to
103      * @throws UnsupportedEncodingException If there was a problem working with the parameters in the specified
104      *                                      character encoding
105      */
106     private HttpUriRequest constructHttpRequest(final AciServerDetails serverDetails, final Set<? extends ActionParameter<?>> parameters) throws EncryptionCodecException, URISyntaxException, UnsupportedEncodingException {
107         LOGGER.trace("constructHttpMethod() called...");
108 
109         // Copy the parameters...
110         final Set<? extends ActionParameter<?>> params = (serverDetails.getEncryptionCodec() == null)
111                 ? parameters
112                 : createEncryptedParameters(serverDetails, parameters);
113 
114         final boolean hasPostParameter = parameters.stream()
115                 .anyMatch(ActionParameter::requiresPostRequest);
116 
117         // If an InputStream parameter has been provided, use a post request regardless
118         if (usePostMethod || hasPostParameter) {
119             return createPostMethod(serverDetails, params);
120         } else {
121             return createGetMethod(serverDetails, params);
122         }
123     }

125     /**
126      * Takes the passed in set of parameters and encrypts them.
127      * @param serverDetails The details of the ACI server the request will be sent to
128      * @param parameters    The parameters to send with the ACI action.
129      * @return A set of encrypted parameters
130      * @throws EncryptionCodecException if something went wrong encrypting the parameters
131      */
132     private Set<? extends ActionParameter<?>> createEncryptedParameters(final AciServerDetails serverDetails, final Set<? extends ActionParameter<?>> parameters) throws EncryptionCodecException {
133         LOGGER.trace("createEncryptedParameters() called...");
134 
135         // Generate the query String and put it through the codec...
136         final String data = EncryptionCodecUtils.getInstance().encrypt(
137                 serverDetails.getEncryptionCodec(),
138                 convertParameters(parameters, serverDetails.getCharsetName()),
139                 serverDetails.getCharsetName()
140         );
141 
142         // Create the parameters for an encrypted action...
143         return new ActionParameters(
144                 new AciParameter(AciConstants.PARAM_ACTION, AciConstants.ACTION_ENCRYPTED),
145                 new AciParameter(AciConstants.PARAM_DATA, data)
146         );
147     }

149     /**
150      * Create a {@code GetMethod} and adds the ACI parameters to the query string.
151      * @param serverDetails The details of the ACI server the request will be sent to
152      * @param parameters    The parameters to send with the ACI action.
153      * @return a {@code HttpGet} that is ready to execute the ACI action.
154      * @throws URISyntaxException If there was a problem construction the request URI from the <tt>serverDetails</tt>
155      *                            and <tt>parameters</tt>
156      */
157     private HttpUriRequest createGetMethod(final AciServerDetails serverDetails, final Set<? extends ActionParameter<?>> parameters) throws URISyntaxException {
158         LOGGER.trace("createGetMethod() called...");
159 
160         // Create the URI to use...
161         final URI uri = new URIBuilder()
162                 .setScheme(serverDetails.getProtocol().toString().toLowerCase(Locale.ENGLISH))
163                 .setHost(serverDetails.getHost())
164                 .setPort(serverDetails.getPort())
165                 .setPath("/")
166                 .setQuery(convertParameters(parameters, serverDetails.getCharsetName()))
167                 .build();
168 
169         // Return the constructed get method...
170         return new HttpGet(uri);
171     }

173     /**
174      * Create a {@code PostMethod} and adds the ACI parameters to the request body.
175      * @param serverDetails The details of the ACI server the request will be sent to
176      * @param parameters    The parameters to send with the ACI action.
177      * @return An {@code HttpPost} that is ready to execute the ACI action.
178      * @throws UnsupportedEncodingException Will be thrown if <tt>serverDetails.getCharsetName()</tt> returns a
179      *                                      charset that is not supported by the JVM
180      * @throws URISyntaxException           If there was a problem construction the request URI from the
181      *                                      <tt>serverDetails</tt> and <tt>parameters</tt>
182      */
183     private HttpUriRequest createPostMethod(final AciServerDetails serverDetails, final Set<? extends ActionParameter<?>> parameters) throws URISyntaxException, UnsupportedEncodingException {
184         LOGGER.trace("createPostMethod() called...");
185 
186         // Create the URI to use...
187         final URI uri = new URIBuilder()
188                 .setScheme(serverDetails.getProtocol().toString().toLowerCase(Locale.ENGLISH))
189                 .setHost(serverDetails.getHost())
190                 .setPort(serverDetails.getPort())
191                 .setPath("/")
192                 .build();
193 
194         // Create the method...
195         final HttpPost method = new HttpPost(uri);
196 
197         final Charset charset = Charset.forName(serverDetails.getCharsetName());
198 
199         final boolean requiresMultipart = parameters.stream()
200                 .anyMatch(ActionParameter::requiresPostRequest);
201 
202         if (requiresMultipart) {
203             final MultipartEntityBuilder multipartEntityBuilder = MultipartEntityBuilder.create();
204             multipartEntityBuilder.setCharset(charset);
205 
206             parameters.forEach(parameter -> parameter.addToEntity(multipartEntityBuilder, charset));
207 
208             // Convert the parameters into an entity...
209             method.setEntity(multipartEntityBuilder.build());
210         }
211         else {
212             method.setEntity(
213                     new StringEntity(
214                             convertParameters(parameters, serverDetails.getCharsetName()),
215                             serverDetails.getCharsetName()
216                     )
217             );
218         }
219 
220         // Return the method...
221         return method;
222     }

224     /**
225      * Converts a list of {@code AciParameter} objects into an array of {@code NameValuePair} objects suitable for use
226      * in both POST and GET methods.
227      * @param parameters  The set of parameters to convert.
228      * @param charsetName The name of the charset to use when encoding the parameters
229      * @return an <tt>String</tt> representing the query string portion of a URI
230      */
231     private String convertParameters(final Set<? extends ActionParameter<?>> parameters, final String charsetName) {
232         LOGGER.trace("convertParameters() called...");
233 
234         // Just incase, remove the allowed null entry...
235         parameters.remove(null);
236 
237         final List<NameValuePair> pairs = new ArrayList<>(parameters.size());
238 
239         LOGGER.debug("Converting {} parameters...", parameters.size());
240 
241         NameValuePair actionPair = null;
242 
243         for (final ActionParameter<?> parameter : parameters) {
244             final Object value = parameter.getValue();
245 
246             if (value instanceof String) {
247                 final String stringValue = (String) value;
248 
249                 if (AciConstants.PARAM_ACTION.equalsIgnoreCase(parameter.getName())) {
250                     actionPair = new BasicNameValuePair(parameter.getName(), stringValue);
251                 } else {
252                     pairs.add(new BasicNameValuePair(parameter.getName(), stringValue));
253                 }
254             }
255         }
256 
257         // Ensure that the action=XXX parameter is the first thing in the list...
258         Validate.isTrue(actionPair != null, "No action parameter found in parameter set, please set one before trying to execute an ACI request.");
259         pairs.add(0, actionPair);
260 
261         // Convert to a string and return...
262         return URLEncodedUtils.format(pairs, charsetName);
263     }

265     /**
266      * Execute an ACI action on the specific ACI server.
267      * @param serverDetails Details of the ACI server to send the action to
268      * @param parameters    The parameters to send with the ACI action
269      * @return An <tt>AciResponseInputStream</tt> containing the ACI response
270      * @throws IOException              If an I/O (transport) error occurs. Some transport exceptions can be recovered from
271      * @throws AciHttpException         If a protocol exception occurs. Usually protocol exceptions cannot be recovered from
272      * @throws IllegalArgumentException if the <tt>httpClient</tt> property is <tt>null</tt> or <tt>parameters</tt> is <tt>null</tt>
273      */
274     @Override
275     public AciResponseInputStream executeAction(final AciServerDetails serverDetails, final Set<? extends ActionParameter<?>> parameters) throws IOException, AciHttpException {
276         LOGGER.trace("executeAction() called...");
277 
278         Validate.notNull(httpClient, "You must set the HttpClient instance to use before using this class.");
279         Validate.notEmpty(parameters, "The parameter set must not be null or empty.");
280 
281         try {
282             // Create the method we're going to use...
283             final HttpUriRequest request = constructHttpRequest(serverDetails, parameters);
284 
285             LOGGER.debug("Executing action on {}:{}...", serverDetails.getHost(), serverDetails.getPort());
286 
287             // Execute the method...
288             final HttpResponse response = httpClient.execute(request);
289 
290             final int statusCode = response.getStatusLine().getStatusCode();
291             LOGGER.debug("Executed method and got status code - {}...", statusCode);
292 
293             // Treat anything other than a 2xx status code as an error...
294             if ((statusCode < 200) || (statusCode >= 300)) {
295                 // close the connection so it can be reused
296                 EntityUtils.consume(response.getEntity());
297 
298                 throw new AciHttpException("The server returned a status code, " + statusCode + ", that wasn't in the 2xx Success range.");
299             }
300 
301             // Decorate the InputStream so we can release the HTTP connection once the stream's been read...
302             return decryptResponse(serverDetails.getEncryptionCodec(), response)
303                     ? new DecryptingAciResponseInputStreamImpl(serverDetails, response)
304                     : new AciResponseInputStreamImpl(response);
305         } catch (final ClientProtocolException cpe) {
306             throw new AciHttpException("A HTTP protocol Exception has been caught while trying to execute the ACI request.", cpe);
307         } catch (final EncryptionCodecException ece) {
308             throw new AciHttpException("Unable to send the ACI request due to an encryption failure.", ece);
309         } catch (final URISyntaxException urise) {
310             throw new AciHttpException("Unable to construct the URI required to send the ACI request.", urise);
311         }
312     }

314     private boolean decryptResponse(final EncryptionCodec encryptionCodec, final HttpResponse response) {
315         LOGGER.trace("decryptResponse() called...");
316 
317         // If there is no encryptionCodec then we don't need to check the headers...
318         boolean decryptResponse = (encryptionCodec != null);
319 
320         LOGGER.debug("Using an EncryptionCodec - {}...", decryptResponse);
321 
322         if (decryptResponse) {
323             LOGGER.debug("Checking AUTN-Content-Type response header...");
324 
325             // This response header is only supplied with encrypted responses, so if it's not there we don't decrypt,
326             // i.e. it's either an OEM IDOL, or they did encryptResponse=false...
327             final Header header = response.getFirstHeader("AUTN-Content-Type");
328             if (header == null) {
329                 LOGGER.debug("No AUTN-Content-Type response header, so don't auto decrypt response...");
330                 decryptResponse = false;
331             }
332         }
333 
334         // Send back the flag...
335         return decryptResponse;
336     }
