8   import com.autonomy.aci.client.transport.AciServerDetails;
9   import com.autonomy.aci.client.transport.ActionParameter;
10  
11  import java.util.Set;

13  /**
14   * Defines methods for executing ACI actions on any ACI server and processing of the resulting response.
15   */
16  public interface AciService {
17  
18      /**
19       * Executes an ACI action and processes the response with the supplied <tt>Processor</tt>.
20       * @param <T> Return type.
21       * @param parameters The parameters to use with the ACI command. This <strong>should</strong> include an {@code
22       *                   Action=<command>} parameter
23       * @param processor  The <tt>Processor</tt> to use for converting the response stream into an object
24       * @return The ACI response encoded as an object of type <tt>T</tt>
25       * @throws AciServiceException If an error occurred during the communication with the ACI Server, processing the
26       *                             response or if the response contained an error
27       */
28      <T> T executeAction(Set<? extends ActionParameter<?>> parameters, Processor<T> processor);
29  
30      /**
31       * Executes an ACI action and processes the response with the supplied <tt>Processor</tt>.
32       * @param <T> Return type.
33       * @param serverDetails The connection details of the ACI Server to execute the action on
34       * @param parameters    The parameters to use with the ACI command. This <strong>should</strong> include an {@code
35       *                      Action=<command>} parameter
36       * @param processor     The <tt>Processor</tt> to use for converting the response stream into an object
37       * @return The ACI response encoded as an object of type <tt>T</tt>
38       * @throws AciServiceException If an error occurred during the communication with
39       *                             the ACI Server, processing the response or if the response contained an error
40       */
41      <T> T executeAction(AciServerDetails serverDetails, Set<? extends ActionParameter<?>> parameters, Processor<T> processor);
42  
43  }
