29  /**
30   * <tt>Processor</tt> implementation that converts an ACI response into a DOM <tt>Document</tt> for further processing
31   * either directly via DOM or via XPath.
32   */
33  public class DocumentProcessor implements Processor<Document> {
34  
35      private static final long serialVersionUID = -1757174558649421058L;
36  
37      private static final Logger LOGGER = LoggerFactory.getLogger(DocumentProcessor.class);
38  
39      /**
40       * Converts the <tt>InputStream</tt> into a DOM <tt>Document</tt>.
41       * @param response The <tt>InputStream to convert</tt>
42       * @return The resulting DOM <tt>Document</tt>
43       * @throws ProcessorException If there was an exception while parsing the input stream or configuring the
44       *                            <tt>DocumentBuilderFactory</tt>.
45       */
46      private Document convertACIResponseToDOM(final InputStream response) {
47          LOGGER.trace("convertACIResponseToDOM() called...");
48  
49          try {
50              // Create the factory...
51              final DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
52  
53              // Should speed things up a bit...
54              factory.setNamespaceAware(false);
55              factory.setValidating(false);
56  
57              LOGGER.debug("Converting ACI response to DOM Document...");
58  
59              // Get a document builder and convert the document...
60              return factory.newDocumentBuilder().parse(new InputSource(response));
61          } catch (final ParserConfigurationException | IOException | SAXException pce) {
62              throw new ProcessorException("Unable to parse the ACI response into a DOM document.", pce);
63          }
64      }
65  
66      /**
67       * Checks the DOM <tt>Document</tt> to see if it is an ACI Server error response. If it is, it pulls all the
68       * information contained in the response into an <tt>AciErrorException</tt> and throws it, otherwise it does
69       * nothing.
70       * @param response The DOM <tt>Document</tt> to check.
71       * @throws AciErrorException  If an error response was detected.
72       * @throws ProcessorException If there was any other kind of exception caught during processing.
73       */
74      private void checkACIResponseForError(final Document response) {
75          LOGGER.trace("checkACIResponseForError() called...");
76  
77          // Create an xpath object for getting stuff with...
78          final XPath xpath = XPathFactory.newInstance().newXPath();
79  
80          try {
81              // Get the value of the <response> tag...
82              final String value = xpath.evaluate("/autnresponse/response", response);
83  
84              LOGGER.debug("Response tag has value - {}...", value);
85  
86              if ("ERROR".equals(value)) {
87                  LOGGER.debug("Error response detected, creating an AciErrorException...");
88  
89                  // Create an exception to throw...
90                  final AciErrorException error = new AciErrorException();
91  
92                  // Get the error properties...
93                  error.setErrorId(xpath.evaluate("/autnresponse/responsedata/error/errorid", response));
94                  error.setRawErrorId(xpath.evaluate("/autnresponse/responsedata/error/rawerrorid", response));
95                  error.setErrorString(xpath.evaluate("/autnresponse/responsedata/error/errorstring", response));
96                  error.setErrorDescription(xpath.evaluate("/autnresponse/responsedata/error/errordescription", response));
97                  error.setErrorCode(xpath.evaluate("/autnresponse/responsedata/error/errorcode", response));
98                  try {
99                      error.setErrorTime(DateTimeUtils.getInstance().parseDate(xpath.evaluate("/autnresponse/responsedata/error/errortime", response), "dd MMM yy HH:mm:ss"));
100                 } catch (final ParseException pe) {
101                     LOGGER.error("ParseException caught while trying to convert errortime tag into java.util.Date.", pe);
102                 }
103 
104                 // Throw the error...
105                 throw error;
106             }
107         } catch (final XPathExpressionException xpee) {
108             // Throw a wobbler, as this should never happen...
109             throw new ProcessorException("XPathExpressionException caught while trying to check ACI response for ERROR flag.", xpee);
110         }
111     }
112 
113     /**
114      * Process the ACI response input into a DOM <tt>Document</tt>.
115      * @param aciResponse The ACI response to process
116      * @return A DOM <tt>Document</tt>
117      * @throws AciErrorException  If the ACI response was an error response
118      * @throws ProcessorException If an error occurred during the processing of the ACI response
119      */
120     public Document process(final AciResponseInputStream aciResponse) {
121         LOGGER.trace("process() called...");
122 
123         // Convert the stream into a DOM Document...
124         final Document document = convertACIResponseToDOM(aciResponse);
125 
126         LOGGER.debug("Checking for ERROR response...");
127 
128         // Check the response for an error flag...
129         checkACIResponseForError(document);
130 
131         LOGGER.debug("Returning the DOM Document to caller...");
132 
133         // Return the document...
134         return document;
135     }
