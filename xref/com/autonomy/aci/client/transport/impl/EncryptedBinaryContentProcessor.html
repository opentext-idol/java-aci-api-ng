12  import com.autonomy.aci.client.util.EncryptionCodecUtils;

14  import java.io.ByteArrayInputStream;
15  import java.io.ByteArrayOutputStream;
16  import java.io.IOException;
17  import javax.xml.stream.XMLStreamException;
18  import javax.xml.stream.XMLStreamReader;
19  import javax.xml.stream.events.XMLEvent;
20  
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  class EncryptedBinaryContentProcessor extends AbstractEncryptedResponseProcessor<ByteArrayInputStream> {
25  
26      private static final Logger LOGGER = LoggerFactory.getLogger(EncryptedBinaryContentProcessor.class);
27  
28      private final ByteArrayOutputStream buffer = new ByteArrayOutputStream();
29  
30      public EncryptedBinaryContentProcessor(final EncryptionCodec encryptionCodec, final String charsetName) {
31          super(encryptionCodec, charsetName);
32      }
33  
34      @Override
35      public ByteArrayInputStream process(final XMLStreamReader aciResponse) throws AciErrorException, ProcessorException {
36          try {
37              while (aciResponse.hasNext()) {
38                  // Get the event type...
39                  final int eventType = aciResponse.next();
40  
41                  if ((eventType == XMLEvent.START_ELEMENT) && "autn:encrypteddata".equals(aciResponse.getLocalName())) {
42                      LOGGER.debug("Found an encrypted data block, decrypting and appending to buffer...");
43                      buffer.write(encryptionCodec.decrypt(EncryptionCodecUtils.getInstance().toBytes(aciResponse.getElementText(), charsetName)));
44                  }
45              }
46  
47              // Return the decrypted response...
48              return new ByteArrayInputStream(buffer.toByteArray());
49          } catch (final XMLStreamException xmlse) {
50              throw new ProcessorException("Unable to decrypt the ACI response due to a problem with the input stream.", xmlse);
51          } catch (final IOException ioe) {
52              throw new ProcessorException("Unable to decrypt the ACI response due to an IOException.", ioe);
53          } catch (final EncryptionCodecException ece) {
54              throw new ProcessorException("Unable to decrypt the ACI response.", ece);
55          }
56      }
57  
58  }
