6   package com.autonomy.aci.client.services.impl;
7   
8   import com.autonomy.aci.client.services.AciErrorException;
9   import com.autonomy.aci.client.services.Processor;
10  import com.autonomy.aci.client.transport.AciResponseInputStream;
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  
14  /**
15   * Takes an ACI response and returns it as a byte array. This processor should only be used for those actions that
16   * <strong>don't return XML</strong> by default, for example, cluster actions that return an image. This processor will
17   * correctly handle an XML error response by checking to see what the content type of the returned response is before
18   * processing it.
19   * <p>
20   * By default this processor uses {@link ErrorProcessor} and {@link ByteArrayProcessor} to do it's work. The
21   * implementations to be used can be changed by using the appropriate accessor methods.
22   * <p>
23   * <strong>Note:</strong> If the content type is <tt>text/xml</tt> then an
24   * {@link com.autonomy.aci.client.services.AciErrorException} will be thrown regardless of the contents of the actual
25   * response. If the response didn't contain an error, then the resulting exception's error properties will all be
26   * <tt>null</tt>. Any other content type will result in the response being returned in a <tt>byte[]</tt>.
27   */
28  public class BinaryResponseProcessor implements Processor<byte[]> {
29  
30      private static final long serialVersionUID = 1262978871829721008L;
31  
32      private static final Logger LOGGER = LoggerFactory.getLogger(ByteArrayProcessor.class);
33  
34      /**
35       * This holds the error processor to use if the content type of the response is <tt>text/xml</tt>.
36       */
37      private Processor<AciErrorException> errorProcessor = new ErrorProcessor();
38  
39      /**
40       * This holds the processor to use if the content type of the response isn't <tt>text/xml</tt>.
41       */
42      private Processor<byte[]> byteArrayProcessor = new ByteArrayProcessor();
43  
44      public byte[] process(final AciResponseInputStream inputStream) {
45          LOGGER.trace("process() called...");
46  
47          // Check the content type to determine what we do next...
48          final String contentType = inputStream.getContentType();
49  
50          LOGGER.debug("Content-type is {}...", contentType);
51  
52          if ("text/xml".equalsIgnoreCase(contentType)) {
53              // Process the error response...
54              throw errorProcessor.process(inputStream);
55          } else {
56              // Return the raw bytes...
57              return byteArrayProcessor.process(inputStream);
58          }
59      }
60  
61      public Processor<AciErrorException> getErrorProcessor() {
62          return errorProcessor;
63      }
64  
65      public void setErrorProcessor(final Processor<AciErrorException> errorProcessor) {
66          this.errorProcessor = errorProcessor;
67      }
68  
69      public Processor<byte[]> getByteArrayProcessor() {
70          return byteArrayProcessor;
71      }
72  
73      public void setByteArrayProcessor(final Processor<byte[]> byteArrayProcessor) {
74          this.byteArrayProcessor = byteArrayProcessor;
75      }
76  
77  }
