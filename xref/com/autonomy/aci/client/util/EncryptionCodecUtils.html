6   package com.autonomy.aci.client.util;
7   
8   import com.autonomy.aci.client.transport.EncryptionCodec;
9   import com.autonomy.aci.client.transport.EncryptionCodecException;
10  
11  import java.io.UnsupportedEncodingException;
12  
13  /**
14   * This is a utility class that encrypts and decrypts <tt>String</tt> objects, it main function is to hide the required
15   * conversions between strings and byte arrays.
16   */
17  public class EncryptionCodecUtils {
18  
19      // Thread safe singleton instance of our class.
20      private static final EncryptionCodecUtils INSTANCE = new EncryptionCodecUtils();
21  
22      /**
23       * Returns the thread safe singleton instance of this class.
24       * @return The thread safe singleton instance of this class
25       */
26      public static EncryptionCodecUtils getInstance() {
27          return INSTANCE;
28      }
29  
30      /**
31       * Encrypt the given string with the supplied codec and use the supplied charset name for any string/byte
32       * conversions.
33       * @param codec       The <tt>EncryptionCodec</tt> to use
34       * @param string      The input to encrypt
35       * @param charsetName The name of charset to use for string and byte array conversion
36       * @return The encrypted input
37       * @throws EncryptionCodecException if something went wrong during either the conversion of encrypting
38       */
39      public String encrypt(final EncryptionCodec codec, final String string, final String charsetName) throws EncryptionCodecException {
40          final byte[] bytes = toBytes(string, charsetName);
41          final byte[] encrypted = codec.encrypt(bytes);
42          return toString(encrypted, charsetName);
43      }
44  
45      /**
46       * Decrypt the given string with the supplied codec and use the supplied charset name for any string/byte conversions.
47       * @param codec       The <tt>EncryptionCodec</tt> to use
48       * @param string      The input to decrypt
49       * @param charsetName The name of charset to use for string and byte array conversion
50       * @return The decrypted input
51       * @throws EncryptionCodecException if something went wrong during either the conversion of decrypting.
52       */
53      public String decrypt(final EncryptionCodec codec, final String string, final String charsetName) throws EncryptionCodecException {
54          final byte[] bytes = toBytes(string, charsetName);
55          final byte[] decrypted = codec.decrypt(bytes);
56          return toString(decrypted, charsetName);
57      }
58  
59      /**
60       * Constructs a new <tt>String</tt> by decoding the specified array of bytes using the specified charset. The length
61       * of the new <tt>String</tt> is a function of the charset, and hence may not be equal to the length of the byte
62       * array.
63       * <p>
64       * The behavior of this method when the given bytes are not valid in the given charset is unspecified. The {@link
65       * java.nio.charset.CharsetDecoder} class should be used when more control over the decoding process is required.
66       * @param bytes       the bytes to be decoded into characters
67       * @param charsetName the name of a supported {@link java.nio.charset.Charset </code>charset<code>}
68       * @return A string representation of the <tt>bytes</tt> argument
69       * @throws EncryptionCodecException If the named charset is not supported
70       * @throws NullPointerException     If either <tt>bytes</tt> or <tt>charsetName</tt> is <tt>null</tt>
71       */
72      public String toString(final byte[] bytes, final String charsetName) throws EncryptionCodecException {
73          try {
74              return new String(bytes, charsetName);
75          } catch (final UnsupportedEncodingException uee) {
76              throw new EncryptionCodecException("Unable to convert the byte array into a String.", uee);
77          }
78      }
79  
80      /**
81       * Encodes the <tt>string</tt> into a sequence of bytes using the named charset, storing the result into a new byte
82       * array.
83       * <p>
84       * The behavior of this method when this string cannot be encoded in the given charset is unspecified. The {@link
85       * java.nio.charset.CharsetEncoder} class should be used when more control over the encoding process is required.
86       * @param string      The <tt>String</tt> to convert to a byte array.
87       * @param charsetName The name of a supported {@link java.nio.charset.Charset </code>charset<code>}
88       * @return The resultant byte array
89       * @throws EncryptionCodecException If the named charset is not supported
90       * @throws NullPointerException     if either <tt>string</tt> or <tt>charsetName</tt> is <tt>null</tt>
91       */
92      public byte[] toBytes(final String string, final String charsetName) throws EncryptionCodecException {
93          try {
94              return string.getBytes(charsetName);
95          } catch (final UnsupportedEncodingException uee) {
96              throw new EncryptionCodecException("Unable to convert the String into a byte array.", uee);
97          }
98      }
