108         final StringEntity stringEntity = new StringEntity(encryptedBinaryResponse);
109         stringEntity.setContentType("text/xml");
110 
111         httpResponse.setEntity(stringEntity);
112         httpResponse.setHeaders(new Header[]{new BasicHeader("AUTN-Content-Type", "image/jpeg"), new BasicHeader("Content-Type", "text/xml")});
113 
114         // This should decrypt the response...
115         final DecryptingAciResponseInputStreamImpl stream = new DecryptingAciResponseInputStreamImpl(serverDetails, httpResponse);
116 
117         // Create a hash of the binary content so we don't have to check byte for byte...
118         final MessageDigest md = MessageDigest.getInstance("MD5");
119         final String hash = new BigInteger(md.digest(toBytes(stream))).toString(16);
120 
121         assertThat(hash, is(equalTo("-11dd06c110a78fb8c3d2c72ff0289bd3")));
122 
123         // Check that wqe have the right content type...
124         assertThat(stream.getContentType(), is(equalTo("image/jpeg")));
125     }

172     private static String toString(final InputStream inputStream) throws IOException {
