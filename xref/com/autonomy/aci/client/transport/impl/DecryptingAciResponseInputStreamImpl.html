20  /**
21   * This <tt>AciResponseInputStream</tt> implementation decrypts ACI responses from actions that have been sent with the
22   * {@code EncryptResponse} parameter set to {@code true}.
23   */
24  public class DecryptingAciResponseInputStreamImpl extends AciResponseInputStreamImpl {
25  
26      private static final Logger LOGGER = LoggerFactory.getLogger(DecryptingAciResponseInputStreamImpl.class);
27  
28      /**
29       * This is what we'll decrypt the input into...
30       */
31      private final ByteArrayInputStream decryptedResponse;
32  
33      /**
34       * Holds the content type of the encrypted data.
35       */
36      private final String contentType;
37  
38      /**
39       * Creates a new instance of DecryptingAciResponseInputStreamImpl.
40       * @param serverDetails The <tt>AciServerDetails</tt> that contains the <tt>EncryptionCodec</tt> that is being used
41       *                      to decrypt the ACI response and the character encoding being used on this ACI request/response
42       * @param response      An {@code HttpResponse} that contains the ACI response as an {@code InputStream}
43       * @throws IOException If an I/O error occurs
44       */
45      public DecryptingAciResponseInputStreamImpl(final AciServerDetails serverDetails, final HttpResponse response) throws IOException {
46          super(response);
47  
48          try {
49              LOGGER.debug("Checking AUTN-Content-Type header...");
50  
51              // Get the autonomy content type header...
52              final String autnContentType = getHeader("AUTN-Content-Type");
53  
54              // This shouldn't be necessary, but just in case someone is using this outwith the API...
55              if (StringUtils.isBlank(autnContentType)) {
56                  LOGGER.debug("AUTN-Content-Type header doesn't exist, will return response unprocessed...");
57  
58                  // Copy the response to the decryptedResponse property unmolested... 
59                  final ByteArrayOutputStream buffer = new ByteArrayOutputStream();
60                  IOUtils.getInstance().copy(new AciResponseInputStreamImpl(response), buffer);
61                  decryptedResponse = new ByteArrayInputStream(buffer.toByteArray());
62  
63                  // Set the content type...
64                  contentType = super.getContentType();
65              } else {
66                  LOGGER.debug("AUTN-Content-Type header is {}...", autnContentType);
67  
68                  // Set the content type...
69                  contentType = autnContentType;
70  
71                  // Get the response and decrypt it into the internal buffer... We have to create a new
72                  // AciResponseInputStreamImpl rather than giving this, otherwise the overridden methods we be called
73                  // resulting in bad things happening...
74                  decryptedResponse = "text/xml".equals(autnContentType)
75                          ? new EncryptedTextContentProcessor(serverDetails.getEncryptionCodec(), serverDetails.getCharsetName()).process(in)
76                          : new EncryptedBinaryContentProcessor(serverDetails.getEncryptionCodec(), serverDetails.getCharsetName()).process(in);
77              }
78          } catch (final ProcessorException pe) {
79              LOGGER.error("ProcessorException caught while trying to decrypt the ACI response", pe);
80              throw new IOException(pe.getMessage());
81          }
82      }
83  
84      /**
85       * Return the content type of the response. Most likely to be <tt>text/xml</tt>, but could be <tt>image/jpeg</tt> if
86       * the response if from a <tt>ClusterServe2DMap</tt> action, for example.
87       * @return The content type of the response, if no Content-Type header was found, then <tt>null</tt> is returned
88       */
89      @Override
90      public String getContentType() {
91          return contentType;
92      }
93  
94      @Override
95      public int read() throws IOException {
96          return decryptedResponse.read();
97      }
98  
99      @Override
100     public int read(final byte[] bytes, final int off, final int len) throws IOException {
101         return decryptedResponse.read(bytes, off, len);
102     }
103 
104     @Override
105     public long skip(final long num) throws IOException {
106         return decryptedResponse.skip(num);
107     }
108 
109     @Override
110     public int available() throws IOException {
111         return decryptedResponse.available();
112     }
113 
114     @Override
115     public synchronized void mark(final int readlimit) {
116         decryptedResponse.mark(readlimit);
117     }
118 
119     @Override
120     public synchronized void reset() throws IOException {
121         decryptedResponse.reset();
122     }
123 
124     @Override
125     public boolean markSupported() {
126         return decryptedResponse.markSupported();
127     }
