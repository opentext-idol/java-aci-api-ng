27      private String constructString(final int[] unicodeChars) {
28          final StringBuilder buffer = new StringBuilder();
29          if (unicodeChars != null) {
30              for (final int unicodeChar : unicodeChars) {
31                  buffer.append((char) unicodeChar);
32              }
33          }
34          return buffer.toString();
35      }

55      @Test
56      public void testUTF8RoundTrip() {
57          // Get the codec...
58          final AciURLCodec codec = AciURLCodec.getInstance();
59  
60          final String ru_msg = constructString(RUSSIAN_STUFF_UNICODE);
61          final String ch_msg = constructString(SWISS_GERMAN_STUFF_UNICODE);
62  
63          assertEquals("%D0%92%D1%81%D0%B5%D0%BC_%D0%BF%D1%80%D0%B8%D0%B2%D0%B5%D1%82", codec.encode(ru_msg));
64          assertEquals("Gr%C3%BCezi_z%C3%A4m%C3%A4", codec.encode(ch_msg));
65  
66          assertEquals(ru_msg, codec.decode(codec.encode(ru_msg)));
67          assertEquals(ch_msg, codec.decode(codec.encode(ch_msg)));
68      }
