6   package com.autonomy.aci.client.transport;
7   
8   import java.io.Serializable;
9   
10  /**
11   * IDOL Server supports a number of different methods for encrypting ACI actions between itself and the client. This
12   * interface defines the methods that the {@code AciHttpClient} will call to encrypt and decrypt an ACI action.
13   * <p>
14   * Classes that implement this interface should be aware of just how IDOL expects the incoming ACI action to be
15   * formatted. If the user is sending a query, for example, the action parameters sent may look something like:
16   * <pre>
17   *   action=query&amp;text=tour%20de%20france&amp;maxresults=10&amp;combine=simple&amp;print=all
18   * </pre>
19   * When the {@code AciHttpClient} detects that an {@code EncryptionCodec} should be used, it creates a new set of ACI
20   * parameters and uses the output of the codec as the value to one of those parameters.
21   * <p>
22   * When using a {@code Cipher} to encrypt text, the result is a {@code byte[]} not a string. HTTP is a text based
23   * protocol, so the {@code EncryptionCodec} needs to convert the {@code byte[]} to a {@code String} by Base64 encoding
24   * it. Furthermore, as query strings can be quite long, even before encrypting. Thus the query string that is sent to
25   * IDOL will look it:
26   * <pre>
27   *   action=encrypted&amp;data=&lt;base64(encrypt(deflate(original query string)))&gt;
28   * </pre>
29   * When IDOL sends the response back, the {@code AciHttpClient} takes the contents of the
30   * <tt>/autnresponse/responsedata/autn:encrypted</tt> element and applies the steps to encrypt in reverse, i.e. base64
31   * decode, decrypt and inflate. This will then result in a {@code byte[]} that contains the actual ACI response to be
32   * sent back via the {@code AciService}.
33   * <p>
34   * The reason the codec works with byte arrays and not Strings, is due to IDOL Server having a number of actions that
35   * return binary data, i.e. <tt>action=ClusterServe2DMap</tt>. There is a utility class {@link
36   * com.autonomy.aci.client.util.EncryptionCodecUtils} that has methods to make it easier to encrypt/decrypt <tt>String</tt>
37   * objects.
38   */
39  public interface EncryptionCodec extends Serializable {
40  
41      /**
42       * This method should firstly deflate {@code bytes}, then encrypt it and finally Base64 encode it.
43       * @param bytes An array of bytes to encrypt.
44       * @return A Base64 encoded, encrypted, deflated copy of the input array.
45       * @throws EncryptionCodecException if there was a problem during any of the three stages of processing.
46       */
47      byte[] encrypt(byte[] bytes) throws EncryptionCodecException;
48  
49      /**
50       * This method should firstly Base64 decode {@code bytes}, then decrypt it and finally inflate it.
51       * @param bytes A Base64 encoded, encrypted and deflated array of bytes.
52       * @return The original unencrypted content as a byte array.
53       * @throws EncryptionCodecException if there was a problem during any of the three stages of processing.
54       */
55      byte[] decrypt(byte[] bytes) throws EncryptionCodecException;
56  
57  }
